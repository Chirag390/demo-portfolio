#!/bin/sh

COMMIT_MSG_FILE=$1
COMMIT_MSG=$(cat "$COMMIT_MSG_FILE")

# Rule: Must start with JIRA-ID like ABC-123
PATTERN="^[A-Z]+-[0-9]+: .+"

if ! echo "$COMMIT_MSG" | grep -Eq "$PATTERN"; then
  echo "‚ùå Invalid commit message!"
  echo "‚úÖ Use format: JIRA-123: message"
  exit 1
fi

echo "‚úÖ commit-msg passed!"
exit 0
